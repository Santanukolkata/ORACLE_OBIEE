Step 1:
select * from
(select G_MONTH_NAME,
  MALE,
  FEMALE
from EMP_COUNT_MONTH_WISE
)
UNPIVOT (CNT FOR GENDER IN (MALE,FEMALE));

Final Step:

select * from
(select G_MONTH_NAME,
  MALE,
  FEMALE
from EMP_COUNT_MONTH_WISE)
UNPIVOT (CNT FOR GENDER IN (MALE,FEMALE))
PIVOT (SUM(CNT) FOR (G_MONTH_NAME) IN('March' AS "MAR",'February' AS "FEB",'January' AS "JAN",'November' AS "NOV",'October' AS "OCT",'September' AS "SEP"));
